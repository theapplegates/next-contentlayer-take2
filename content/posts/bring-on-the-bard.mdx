import React, { useState } from "react";
import Image from "next/image";

const App = () => {
	const [photos, setPhotos] = useState([])}

	const fetchPhotos = async () => {
		const response = await fetch("https://api.unsplash.com/photos?client_id=HD8UbF03FpBGniqSbmC47gYO2PuHQfsnL9te9k5iYlY&per_page=10");
		const data = await response.json();
		setPhotos(data);
	};

	useEffect(() => {
		fetchPhotos();
	}, []);

	return (
		<div>
			{photos.map((photo) => (
				<div key={photo.id}>
					<Image src={photo.url} alt={photo.description} />
					<p>Photo by <a href={photo.user.links.html} target="_blank">
						{photo.user.name}
					</a> on <a href="https://unsplash.com" target="_blank">Unsplash</a></p>
				</div>
			))}
		</div>
	);
};

}
]
}
export default App;
